# Rodando o projeto.

# Front-end

    -$ cd frontend
    -$ yarn || npm install
    -$ yarn start  || npm start

    Disponivel em: http://localhost:3000

# Back-end

    -$ cd backend
    -$ yarn || npm install
    -$ yarn build || npm build
    -$ yarn start || npm start

    Disponivel em: http://localhost:4000/ping

-- Node v18 required.
